import cv2
import numpy as np

# Bilder laden
img1 = cv2.imread(r'C:\Users\i-asya\Desktop\Software\BV-Tutorial\Step3\basis3A.png', 1)
img2 = cv2.imread(r'C:\Users\i-asya\Desktop\Software\BV-Tutorial\Step3\basis3B.png', 1)

# Bilder auf die gleiche Größe skalieren
img1 = cv2.resize(img1, (img2.shape[1], img2.shape[0]))

# Logische AND-Verknüpfung durchführen
result = cv2.bitwise_and(img1, img2)

# Bilder anzeigen
cv2.imshow('Bild 1', img1)
cv2.imshow('Bild 2', img2)
cv2.imshow('Verknüpftes Bild', result)

cv2.waitKey(0)
cv2.destroyAllWindows()

import cv2

# Bild lesen (als Graustufenbild)
img = cv2.imread(r'C:\Users\i-asya\Desktop\bild1.png', cv2.IMREAD_GRAYSCALE)

# Schwellenwert setzen (127 in diesem Fall)
threshold_value = 127

# Anwenden des Schwellenwertverfahrens
#Pixelwerte unter 127 werden zu 0 gesetzt
#Pixelwerte größer 127 werden zu 255 gesetzt
ret, thresholded = cv2.threshold(img, threshold_value, 255, cv2.THRESH_BINARY)

# Anzeigen des Originalbilds und des thresholded Bilds
cv2.imshow('Originalbild', img)
cv2.imshow('Thresholded Bild', thresholded)

cv2.waitKey(0)
cv2.destroyAllWindows()

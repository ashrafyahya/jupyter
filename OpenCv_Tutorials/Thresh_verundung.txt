import cv2
import numpy as np

# Bild lesen (als Graustufenbild)
img = cv2.imread(r'C:\Users\i-asya\Desktop\Software\BV-Tutorial\Step5\basis5.png', cv2.IMREAD_GRAYSCALE)

# Schwellenwert setzen (127 in diesem Fall)
ret, thresholded= cv2.threshold(img, 127, 255, cv2.THRESH_BINARY)

# Invertieren des thresholded Bilds (Kreise werden wei√ü, Hintergrund schwarz)
inverted = cv2.bitwise_not(thresholded)

kernel= np.ones((3,3), np.uint8)
inverted= cv2.dilate(inverted, kernel, iterations=1 )

#Verundung
verundet = cv2.bitwise_and(inverted, img)


# Anzeigen des Originalbilds und des Sobel-Bilds
cv2.imshow('thresholded', inverted)
cv2.imshow('verundet', verundet)

cv2.waitKey(0)
cv2.destroyAllWindows()

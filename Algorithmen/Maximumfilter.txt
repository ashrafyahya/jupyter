def max_filter(input_pic, kernel_size):
    height= len(input_pic)
    width= len(input_pic[0])
    halb_kernel= kernel_size//2
    output_pic= [[0]* width for _ in range(height)]
    
    for y in range(halb_kernel, height- halb_kernel):
        for x in range(halb_kernel, width- halb_kernel):
            bereich=[]
            for ky in range(y- halb_kernel, y+ halb_kernel ):
                for kx in range(x - halb_kernel, x+ halb_kernel):
                    bereich.append(input_pic[ky][kx])
                    
            bereich.sort()
            #Max-Filter nutzt den größten Wert im Bereich
            max_filter= max(bereich)
            output_pic[x][y]= max_filter
    return output_pic
            
kernel_size= 3

input_pic=  [[4, 9, 2, 5],
             [3, 5, 8, 1],
             [7, 2, 6, 0],
             [6, 3, 2, 8]]

filtered_pic= max_filter(input_pic, kernel_size)

print("Input picture:")
for zeile in input_pic:
    print(zeile)

print("Output picture: ")
for zeile in filtered_pic:
    print(zeile)

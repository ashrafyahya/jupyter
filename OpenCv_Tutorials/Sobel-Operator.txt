import cv2
import numpy as np

# Bild lesen (als Graustufenbild)
img = cv2.imread(r'C:\Users\i-asya\Desktop\Software\BV-Tutorial\Step4\basis4.png', cv2.IMREAD_GRAYSCALE)

# Sobel-Operator in der x-Richtung anwenden
#Der Parameter ksize gibt die Größe des Sobel-Kernels an
#1, 0 in X-Richtung, 0, 1 in Y-Richtung
sobel_x = cv2.Sobel(img, cv2.CV_64F, 1, 0, ksize=3)  # 64-Bit float, Ableitung in x-Richtung

# Konvertieren Sie das Ergebnis in ein 8-Bit-Bild
sobel_x = cv2.convertScaleAbs(sobel_x)

# Anzeigen des Originalbilds und des Sobel-Bilds
cv2.imshow('Originalbild', img)
cv2.imshow('Sobel-Bild (x-Richtung)', sobel_x)

cv2.waitKey(0)
cv2.destroyAllWindows()
